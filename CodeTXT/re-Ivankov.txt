/*
Дима, приведенные в отчете тесты демонстрируют правильное выполнение цикла. Молодец.
Но Тестирование корректной работы функций отсутствует.
Что обчзательно надо проверит:
   - работоспособность рекурсии,
   - Корректное выполнение return<  после которого ничего не должно работать.
Привожу пример такого теста:
*/

int param;
int factor() {
    for (int i = 1; param < 2; i++) {
         return 1;
    }
    for (int i = param; i <= param; i++) {
        int tmp = param;
        param = param - 1;
        return tmp * factor(); 
        param  = 1000 * tmp;  // это пример кода который выполняться не должен
    }
}

int  main() {
   param = 5;      
   int res = factor();      // 5! = 120
   param = 4;
   res= res +  factor();          // проверим результат 5! + 4! = 144
   return 0;
   param = 100;          // не выполняется из-за предшествующего return
   return factor();      // 100! не выполняется
}
